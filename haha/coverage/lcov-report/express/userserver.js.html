/**
 * 所有用户列表
 */
const path = require('path')
const fs = require('fs')
let dirName = path.join(__dirname, 'dbuser.json')
let dbuser = require('./dbuser.json')
function getAllUserList() {
    dbuser.user = []
    for (let i = 0; i < 33; i++) {
        dbuser.user.push({
            name:"xzy" + i,
            sex: '男' + i,
            phone: '184311223' + i
        })
    }
    fs.writeFile(dirName, JSON.stringify(dbuser), {encoding:'utf8'}, () => {
        console.log('文件已经被保存')
    })
    return dbuser.user
    
}
function getPageListInfo(page, size) {
    if(typeof page != 'number' || page < 1) {
        return {
            code: 0,
            msg:'page参数类型不合法'
        }
    }
    if(typeof size != 'number' || size < 1) {
        return {
            code: 2,
            msg:'size参数类型不合法'
        }    
    }
    return {
        code: 1,
        msg:'数据请求成功',
        data: dbuser.user.slice((page-1)*size, page*size)
    }
}
module.exports = {
    getAllUserList : getAllUserList,
    getPageListInfo : getPageListInfo
}
